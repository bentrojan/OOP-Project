def play               
        loop do
          @player_number += 1
          @ttt.view_board

          turn

                if @ttt.winner?
                        @ttt.view_board
                        puts @player_number % 2 == 0 ? "\nX is the winner!" : "\nO is the winner!"
                        exit
                end          

          if @ttt.cat_game?
                        draw_cat
                        exit
          end
        end
end
